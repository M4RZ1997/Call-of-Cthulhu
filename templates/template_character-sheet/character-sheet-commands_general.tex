% Calculate Scores
\newcommand{\calculateCharacteristicStats}{%
	\StrengthScore{\intcalcAdd{\intcalcMul{\StrengthRolledValue}{5}}{\StrengthModificationValue}}
	\ConstitutionScore{\intcalcAdd{\intcalcMul{\ConstitutionRolledValue}{5}}{\ConstitutionModificationValue}}
	\DexterityScore{\intcalcAdd{\intcalcMul{\DexterityRolledValue}{5}}{\DexterityModificationValue}}
	\IntelligenceScore{\intcalcAdd{\intcalcMul{\IntelligenceRolledValue}{5}}{\IntelligenceModificationValue}}
	\SizeScore{\intcalcAdd{\intcalcMul{\SizeRolledValue}{5}}{\SizeModificationValue}}
	\PowerScore{\intcalcAdd{\intcalcMul{\PowerRolledValue}{5}}{\PowerModificationValue}}
	
	\calculateAppearanceScore{}
	
	\EducationScore{\intcalcAdd{\intcalcMul{\EducationRolledValue}{5}}{\EducationModificationValue}}
	\LuckStarting{\intcalcAdd{\intcalcMul{\LuckRolledValue}{5}}{\LuckModificationValue}}
	
	\HitPointsMaximum{\intcalcDiv{\intcalcAdd{\ConstitutionScoreValue}{\SizeScoreValue}}{10}}
	\MagicPointsMaximum{\intcalcDiv{\PowerScoreValue}{5}}
	\SanityStarting{\PowerScoreValue}
	
	\calculateDamageBonusAndBuild{\intcalcAdd{\StrengthScoreValue}{\SizeScoreValue}}
	\calculateMove{}
}%

\newcommand{\calculateAppearanceScore}{%
	\def\ageDeduction{0}
	\ifthenelse{\CharacterAgeValue < 40}{}{%
		\ifthenelse{\CharacterAgeValue < 50}{
			\def\ageDeduction{5}
		}{%
			\ifthenelse{\CharacterAgeValue < 60}{
				\def\ageDeduction{10}
			}{%
				\ifthenelse{\CharacterAgeValue < 70}{
					\def\ageDeduction{15}
				}{%
					\ifthenelse{\CharacterAgeValue < 80}{
						\def\ageDeduction{20}
					}{%
						\def\ageDeduction{25}
					}%
				}%
			}%
		}%
	}%
	
	\AppearanceScore{\intcalcSub{\intcalcAdd{\intcalcMul{\AppearanceRolledValue}{5}}{\AppearanceModificationValue}}{\ageDeduction}}
	
}%
\newcommand{\calculateDamageBonusAndBuild}[1]{%
	\ifthenelse{#1<64}{%
		\DamageBonus{-2}
		\Build{-2}
	}{%
		\ifthenelse{#1<84}{%
			\DamageBonus{-1}
			\Build{-1}
		}{%
			\ifthenelse{#1<124}{%
				\DamageBonus{None}
				\Build{0}
			}{%
				\ifthenelse{#1<164}{%
					\DamageBonus{+1D4}
					\Build{+1}
				}{%
					\DamageBonus{+1D6}
					\Build{+2}
				}%
			}%
		}%
	}%
}%
\newcommand{\calculateMove}{%
	\ifthenelse{\DexterityScoreValue < \SizeScoreValue}{%
		\ifthenelse{\StrengthScoreValue < \SizeScoreValue}{\def\base{7}}{\def\base{8}}
	}{%
		\ifthenelse{\DexterityScoreValue = \SizeScoreValue}{%
			\def\base{8}
		}{%
			\ifthenelse{\StrengthScoreValue < \SizeScoreValue}{\def\base{8}}{%
				\ifthenelse{\StrengthScoreValue = \SizeScoreValue}{\def\base{8}}{%
					\def\base{9}
				}%
			}
		}
	}%
	\ifthenelse{\CharacterAgeValue < 40}{\def\reduction{0}}{%
		\ifthenelse{\CharacterAgeValue < 50}{\def\reduction{1}}{%
			\ifthenelse{\CharacterAgeValue < 60}{\def\reduction{2}}{%
				\ifthenelse{\CharacterAgeValue < 70}{\def\reduction{3}}{%
					\ifthenelse{\CharacterAgeValue < 80}{\def\reduction{4}}{\def\reduction{5}}%
				}%
			}%
		}%
	}%
	\Move{\intcalcSub{\base}{\reduction}}%
}%

%% Spending Level
\newcommand{\calculateSpendingLevel}{%
	\npthousandsep{'}
	\ifthenelse{\CreditRatingValue < 1}{%
		\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ 0.50}}{\SpendingLevel{\$ 10}}
	}{%
		\ifthenelse{\CreditRatingValue < 10}{%
			\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ 2}}{\SpendingLevel{\$ 40}}
		}{%
			\ifthenelse{\CreditRatingValue < 50}{%
				\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ 10}}{\SpendingLevel{\$ 200}}
			}{%
				\ifthenelse{\CreditRatingValue < 90}{%
					\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ 50}}{\SpendingLevel{\$ \np{1000}}}
				}{%
					\ifthenelse{\CreditRatingValue < 99}{%
						\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ 250}}{\SpendingLevel{\$ \np{5000}}}
					}{%
						\ifthenelse{\equal{\ERA}{1920s}}{\SpendingLevel{\$ \np{5000}}}{\SpendingLevel{\$ \np{100000}}}
					}%
				}%
			}%
		}%
	}%
}%
%% Cash
\newcommand{\calculateCash}{%
	\npthousandsep{'}
	\ifthenelse{\CreditRatingValue < 1}{%
		\ifthenelse{\equal{\ERA}{1920s}}{\Cash{\$ 0.50}}{\Cash{\$ 10}}%
	}{%
		\ifthenelse{\CreditRatingValue < 10}{%
			\ifthenelse{\equal{\ERA}{1920s}}{%
				\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{1}}}%
			}{%
				\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{20}}}%
			}%
		}{%
			\ifthenelse{\CreditRatingValue < 50}{%
				\ifthenelse{\equal{\ERA}{1920s}}{%
					\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{2}}}%
				}{%
					\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{40}}}%
				}%
			}{%
				\ifthenelse{\CreditRatingValue < 90}{%
					\ifthenelse{\equal{\ERA}{1920s}}{%
						\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{5}}}%
					}{%
						\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{100}}}%
					}%
				}{%
					\ifthenelse{\CreditRatingValue < 99}{%
						\ifthenelse{\equal{\ERA}{1920s}}{%
							\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{20}}}%
						}{%
							\Cash{\$ \np{\intcalcMul{\CreditRatingValue}{400}}}%
						}%
					}{%
						\ifthenelse{\equal{\ERA}{1920s}}{\Cash{\$ \np{50000}}}{\Cash{\$ \np{1000000}}}%
					}%
				}%
			}%
		}%
	}%
}%
